Expression = Term *( ( '+' / '-' ) Term )

Term = Power *( ( '*' / '/' ) Power )

Power = Factor *[ '^' Power ]

Factor = Number / Variable / '(' Expression ')'



Number = 1*DIGIT [ ( "," ) 1*DIGIT]

Variable = 'a' / 'b' / 'c' / 'd' / 'e' / 'f' / 'g' / 'h' / 'i' / 'j' / 'k' / 'l' / 'm' / 'n' / 'o' / 'p' / 'q' / 'r' / 's' / 't' / 'u' / 'v' / 'w' / 'x' / 'y' / 'z'

DIGIT = '1' / '2' / '3' / '4' / '5' / '6' / '7' / '8' / '9' / '0'
